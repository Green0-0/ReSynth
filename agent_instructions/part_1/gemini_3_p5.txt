exhaustive_last_names = df['name'].dropna().str.title().tolist()

# 4. Define our uniform generation functions
def generate_exhaustive_male():
    # Every single name has exactly a 1-in-162,254 chance of being picked
    first = random.choice(male_first_names)
    last = random.choice(exhaustive_last_names)
    return f"{first} {last}"

def generate_exhaustive_female():
    first = random.choice(female_first_names)
    last = random.choice(exhaustive_last_names)
    return f"{first} {last}"
